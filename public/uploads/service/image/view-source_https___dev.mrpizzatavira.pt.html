function openCheckoutModal() {

function updateCheckoutModal() {
const cartItems = JSON.parse(localStorage.getItem('cartItems')) || [];
const tableBody = document.getElementById('checkout-table-body');
tableBody.innerHTML = '';

cartItems.forEach(item => {
const row = document.createElement('tr');
row.innerHTML = `
<td>${item.name}</td>
<td>${item.quantity}</td>
<td>${item.price}</td>
<td>${(item.quantity * item.price).toFixed(2)}</td>
<td>${item.additionalDetails}</td>
<td>
<button onclick="removeFromCart(${item.starter_id})">Remove</button>
</td>
`;
tableBody.appendChild(row);
});

const totalPrice = cartItems.reduce((total, item) => total + (item.quantity * item.price), 0);
document.getElementById('total_price').value = totalPrice.toFixed(2);
}

function addToCart(item) {
const existingItem = itemsInCart.find(cartItem => cartItem.starter_id === item.starter_id);

if (existingItem) {
existingItem.quantity += item.quantity;
existingItem.additionalDetails = additionalDetails;
} else {
itemsInCart.push(item);
}

saveToLocalStorage();
updateCartCount();
updateCheckoutModal();
saveToDatabase(item);
showCartMessage();
}


function saveToDatabase(item) {
fetch('/add-to-cart', {
method: 'POST',
headers: {
'Content-Type': 'application/json',
'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]').getAttribute('content')
},
body: JSON.stringify({ items: [item] })
})
.then(response => {
if (!response.ok) {
throw new Error('Network response was not ok');
}
return response.json();
})
.then(data => {
if (data.status !== 'success') {
alert('Failed to save item to database: ' + data.message);
}
})
.catch(error => {
console.error("Error saving item to database:", error);
alert('An error occurred while saving to the database.');
});
}


function removeFromCart(starter_id) {
const itemToRemove = itemsInCart.find(item => item.starter_id === starter_id);

if (itemToRemove) {
itemsInCart = itemsInCart.filter(item => item.starter_id !== starter_id);
saveToLocalStorage();
updateCheckoutModal();
removeFromDatabase(itemToRemove.name);


const distinctItemsCount = itemsInCart.filter(item => item.quantity > 0).length;
showCartMessage(`Removed item from cart. Total: ${distinctItemsCount} item(s) in cart.`);


if (distinctItemsCount === 0) {
setTimeout(() => {
document.getElementById('cartMessage').style.display = 'none';
}, 3000);
}
} else {

showCartMessage('Item not found in cart.');
}
}

function removeFromDatabase(itemName) {
fetch('/remove-from-cart', {
method: 'DELETE',
headers: {
'Content-Type': 'application/json',

},
body: JSON.stringify({ name: itemName })
})
.then(response => {
if (!response.ok) {
throw new Error('Network response was not ok');
}
return response.json();
})
.then(data => {
if (data.status !== 'success') {
console.error("Error removing item from database:", data.message);
}
})
.catch(error => {
console.error("Error removing item from database:", error);
});
}

});

function processOrderDetails(customerName, contactNumber, paymentMethod, csrfToken) {
const items = itemsInCart.map(item => ({
starter_id: item.starter_id,
wines_id: item.wines_id,
dessert_id: item.dessert_id,
pizzalist_id: item.pizzalist_id,
quantity: item.quantity,
name: item.name,
price: item.price,
type: item.type,
additionalDetails: item.additionalDetails || "None"
}));

const totalPrice = items.reduce((total, item) => total + (item.quantity * item.price), 0);
let orderDetails = `• *Customer Name:* ${customerName}\n• *Contact Number:* ${contactNumber}\n• *Payment Method:* ${paymentMethod}\n\n• *Order Details:*\n\n`;

items.forEach(item => {
orderDetails += ` • *${item.name}*\n *Quantity:* ${item.quantity}\n *Price:* $${(item.quantity * item.price).toFixed(2)}\n *Extra:* ${item.additionalDetails || "None"}\n\n`;
});
orderDetails += `• *Total Price:* $${totalPrice.toFixed(2)}\n`;

fetch('/send-whatsapp', {
method: 'POST',
headers: {
'Content-Type': 'application/json',
'X-CSRF-TOKEN': csrfToken
},
body: JSON.stringify({
items: items,
totalPrice: totalPrice.toFixed(2),
customerName: customerName,
contactNumber: contactNumber,
paymentMethod: paymentMethod
})
})
.then(response => response.json())
.then(data => {
if (data.status === 'success') {
itemsInCart = [];
saveToLocalStorage();
updateCheckoutModal();
updateCartProductCount();
localStorage.removeItem('cartItems');
updateCartUI();
document.getElementById('cartMessage').style.display = 'none';

const url = "https://api.whatsapp.com/send?phone=+918320235855&text=" + encodeURIComponent(orderDetails);
window.open(url, '_blank');
alert('Order placed successfully!');
} else {
alert('Failed to process order: ' + data.message);
}
})
.catch(error => {
console.error("Error processing order:", error);
alert('An unexpected error occurred: ' + error.message);
})
.finally(() => {
document.getElementById('orderFormModal').remove();
isSubmitting = false;
});
}
